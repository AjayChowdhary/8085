COMMAND: MOV A,D
	ANI 0F0H;MASKING UPPER 4 BITS AND RS=0 FOR COMMAND
	OUT 02H; PORT C ADDRESS OF 8255
	CALL DELAY
	ORI 02H; ENABLE HIGH AND RS LOW
	CALL DELAY
	OUT 02H
	CALL DELAY
	ANI 0F0H
	OUT 02H
	CALL DELAY
	MOV A,D
	RLC; SHIFTING THE 4 BITS
	RLC
	RLC
	RLC
	ANI 0F0H; MASKING LOWER 4 BITS
	OUT 02H
	CALL DELAY
	ORI 02H; ENABLE HIGH
	CALL DELAY
	OUT 02H
	CALL DELAY
	ANI 0F0H
	OUT 02H
	CALL DELAY
	RET

RST75: MOV A,E; 1 - TEMP, 0 - HEART

RST65: DI
	   PUSH PSW
	   PUSH H
	   MVI A, 0C0H
	   SIM;  OUTPUTTING 1 ON SOD TO CLEAR D F/F

	   MOV A,M; LOADING CURRENT PULSE COUNT
	   INR A
	   STA 0A000H
	   MVI A, 040H
	   SIM; ENABLING D F/F AGAIN
	   EI
	   RET


HEXTOBCD: PUSH PSW
		PUSH D
		MOV C,A              ; GET HEX DATA
         MVI D,00H          ;Clear D for Most significant Byte 
         XRA A             ;Clear Accumulator 

LOOP22:   ADI 01              ;Count the number one by one 
         DAA               ;Adjust for BCD count 
         JNC LOOP21 
         INR D 
LOOP21:   DCR C 
         JNZ LOOP22 
         STA 0A001H          ;Store the Least Significant Byte 
         MOV A,D 
         STA 0A005H          ;Store the Most Significant Byte 
         POP D
         POP PSW
         RET

BCDTOASCII:   MOV B,A   ; Address already stored in HL
        ANI 0F           ;Mask Upper Nibble 
        CALL SUB1        ;Get ASCII code for upper nibble 
        MOV D,A 
        MOV A,B 
        ANI F0           ;Mask Lower Nibble 
        RLC 
        RLC 
        RLC 
        RLC 
        CALL SUB1        ;Get ASCII code for lower nibble 
        MOV C,A
        RET              ;Return from subroutine

SUB1:   CPI 0A
        JC SKIP 
        ADI 07 
SKIP:   ADI 30
        RET              ;Return Subroutine


HEXTOASCII: CALL HEXTOBCD; LSB-A001H, MSB-A005H / HEART RATE ALREADY STORED AT A000H
	
	LDA 0A001H; LOAD HL WITH ADDRESS OF LSB OF HEART RATE
	CALL HEXTOASCII ; LSB - C, MSB - D
	MOV A,C
	STA A002H; STORING MSB 
	MOV A,D
	STA A003H ; STORING LSB

	LDA 0A005H ; LOAD A WITH ADDRESS OF MSB OF HEART RATE
	CALL HEXTOASCII ; LSB - C, MSB - D
	MOV A,C
	STA A006H ; STORING MSB - THIS WILL ALWAYS BE 0 
	MOV A,D
	STA A007H ; STORING LSB
	RET

MEASURE: MVI A,0C0H
	SIM ; MAKING D F/F FUNCTIONAL
	XRA A; A000H - ADDRESS WHERE PULSE COUNT IS STORED
	STA 0A000H ; SETTING PULSE COUNT TO 0 INITIALLY
	EI
	MVI D, 00FH ; CALLING DELAY OF 15 SEC
	CALL DELAYNSEC ; IN WHICH PULSES ARE COUNTED
	DI ; STOP COUNTING PULSES
	MVI A,040H ; MAKE D F/F NON FUNCTIONAL
	SIM

	LDA 0A000H; LOAD A WITH PULSE COUNT
	RLC; MULTIPLY BY 4 TO GET BEATS/MIN
	RLC
	STA 0A000H
	CALL HEXTOASCII
	CALL DATA3
	MVI A,001H; CALL 1 SEC DELAY
	CALL DELAYNSEC
	JMP MEASURE; MEASURE HEART RATE AGAIN
